### 离散化模板

```c++
void discrete(){      // 离散化
    sort(a+1,a+n+1);
    
    for(int i = 1 ; i <= n ; ++i){
        if(i == 1 || a[i] != a[i-1]) {   b[++m] = a[i]; }
    }
    
}
 
 int query(int x)
 {    // 离散化后查找指定元素下标
     return lower_bound(b+1 , b+m+1,x) - b ;
 }
```
