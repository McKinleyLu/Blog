### 模板代码
```java
#include <iostream>
#include <cstring>
using namespace std;
const int N = 1e5+10;
int q[N];
int hh =0 , tt = 0 ;

int h[N],en[N],ne[N],idx = 0 ;
void insert(int x , int y){
    en[idx] = x , ne[idx] = h[x] , h[x] = idx++;
}
void init(){
    memset(h,-1,sizeof h);
}

int n ;
int d[N] ;

int bfs(){
    q[0] = 1 ;
    while(hh <= tt){
        int t = q[hh++];     
        for(int i = t ; i != -1 ; i = ne[i]){
             int j = en[i];
             if(d[j] == -1){
                 d[j] = d[t] + 1 ;
                 q[++ tt] = j ;
             }
        }
    }  
    return d[n];
}

int main(int argc, char const *argv[])
{
    int n ;
    cin >> n ;
    init();
    memset(d,-1,sizeof d);
    for(int i = 1 ; i <= n ; ++i)  {
        int x , y ;
        cin >> x >> y ;
        insert(x,y);
    }

    cout << bfs() <<endl;

    return 0;
}
```
