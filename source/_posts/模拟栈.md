### 模拟栈

实现栈的`push、pop、empty、top`函数。

```java
//2022.3.1 复习
#include <iostream>
#include <string>
using namespace std ;
const int M = 1e5 + 10 ;
int q[M] ;
int m ; 
int tt = -1 ;
bool empty(){
    if(tt == -1) return true ;
    else         return false ;
}
void pop(){
    tt -= 1 ;
}
void push(int x){
    q[tt++] = x ;
}
int  query(){
    return q[tt-1];
}
int main(int argc, char const *argv[])
{
    cin >> m ;
    while(m--){
        string op ;
        cin >> op ;
        if( op == "push"){
            int x ;
            cin >> x ;
            push(x);
        }
        else if(op == "query"){
            cout << query() << endl;
        }
        else if(op == "empty"){
            if(empty())  cout << "YES" << endl;
            else         cout << "NO"<< endl;
        }
        else{
            pop();
        }
    }
    return 0;
}
```
